```python
#!/usr/bin/env python3
"""
Headless Chrome audit using Chrome DevTools Protocol.
Fetches pages with full JavaScript rendering for SEO analysis.
"""

import argparse
import json
import subprocess
import sys
import time
from datetime import datetime
from urllib.parse import urlparse

def get_chrome_port():
    """Get Chrome remote debugging port from session or default."""
    session_file = f"{os.path.expanduser('~')}/.toprank/cdp-sessions/current.json"
    if os.path.exists(session_file):
        with open(session_file) as f:
            session = json.load(f)
            return session.get('port', 9222)
    return 9222

def audit_page(url, port=9222):
    """Audit a page using headless Chrome via CDP."""
    # Use chrome-devtools-cli or direct CDP commands
    # This script uses CDP to navigate and capture rendered HTML
    
    cmd = [
        'chrome-devtools',
        'navigate', url,
        'wait-for', 'networkidle',
        'screenshot', '-',  # Optional: capture screenshot
        'html'  # Get rendered HTML
    ]
    
    result = subprocess.run(cmd, capture_output=True, text=True)
    return {
        'url': url,
        'rendered_html': result.stdout,
        'timestamp': datetime.now().isoformat(),
        'success': result.returncode == 0
    }

def main():
    parser = argparse.ArgumentParser(description='Headless Chrome SEO audit')
    parser.add_argument('--url', required=True, help='URL to audit')
    parser.add_argument('--port', type=int, default=9222, help='Chrome debug port')
    parser.add_argument('--output', help='Output file (JSON)')
    parser.add_argument('--wait', type=int, default=2000, help='Wait ms after load')
    
    args = parser.parse_args()
    
    result = audit_page(args.url, args.port)
    
    if args.output:
        with open(args.output, 'w') as f:
            json.dump(result, f, indent=2)
    else:
        print(json.dumps(result, indent=2))
    
    return 0 if result['success'] else 1

if __name__ == '__main__':
    import os
    sys.exit(main())
```

Create a Python script to detect if a page is a SPA/SSR and requires JavaScript rendering:

```python
#!/usr/bin/env python3
"""
Detect if a page requires JavaScript rendering (SPA/SSR).
Analyzes HTML and response headers to determine rendering needs.
"""

import argparse
import json
import re
import sys

def detect_js_rendering(html_content, headers=None):
    """
    Detect if page likely requires JS rendering.
    
    Returns:
        dict: {
            'requires_js': bool,
            'indicators': list of detected indicators,
            'confidence': 'high' | 'medium' | 'low'
        }
    """
    indicators = []
    
    # High-confidence SPA indicators
    spa_patterns = [
        (r'webpack', 'Uses webpack bundler'),
        (r'react-dom', 'Uses React'),
        (r'vue\.js', 'Uses Vue.js'),
        (r'angular\.js', 'Uses Angular'),
        (r'ng-app', 'AngularJS app'),
        (r'data-reactroot', 'React rendered'),
        (r'id="__next"', 'Next.js app'),
        (r'id="__nuxt"', 'Nuxt app'),
        (r'gatsby-', 'Gatsby site'),
        (r'ember', 'Ember.js'),
    ]
    
    for pattern, desc in spa_patterns:
        if re.search(pattern, html_content, re.IGNORECASE):
            indicators.append(desc)
    
    # Check for SSR indicators (server-rendered, may still need JS)
    ssr_patterns = [
        (r'<html[^>]*ng-version', 'Angular SSR'),
        (r'<!DOCTYPE html>', 'Valid HTML (may be SSR)'),
    ]
    
    for pattern, desc in ssr_patterns:
        if re.search(pattern, html_content, re.IGNORECASE):
            indicators.append(desc)
    
    # Check headers for rendering hints
    if headers:
        if headers.get('x-sveltekit-page'):
            indicators.append('SvelteKit page')
        if headers.get('x-nextjs-cache'):
            indicators.append('Next.js server-rendered')
    
    # Determine confidence
    if len(indicators) >= 2:
        confidence = 'high'
    elif len(indicators) == 1:
        confidence = 'medium'
    else:
        confidence = 'low'
    
    # If any JS framework detected, requires JS
    requires_js = confidence in ('high', 'medium')
    
    return {
        'requires_js': requires_js,
        'indicators': indicators,
        'confidence': confidence
    }

def main():
    parser = argparse.ArgumentParser(description='Detect SPA/SSR rendering needs')
    parser.add_argument('--html', help='HTML content to analyze')
    parser.add_argument('--file', help='File containing HTML')
    parser.add_argument('--output', help='Output JSON file')
    
    args = parser.parse_args()
    
    html_content = args.html
    if args.file:
        with open(args.file) as f:
            html_content = f.read()
    
    if not html_content:
        print("Error: Provide --html or --file", file=sys.stderr)
        return 1
    
    result = detect_js_rendering(html_content)
    
    if args.output:
        with open(args.output, 'w') as f:
            json.dump(result, f, indent=2)
    else:
        print(json.dumps(result, indent=2))
    
    return 0

if __name__ == '__main__':
    sys.exit(main())
```
